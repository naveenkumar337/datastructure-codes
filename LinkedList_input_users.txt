package linkedlist;
import java.util.*;

public class LinkedListDemo {
	
    public static void main(String[] args) {
    	
        Scanner sc=new Scanner(System.in);
        LinkedList<Integer> li=new LinkedList<Integer>();
                    
      while(true) {
    	  System.out.println("1.Add");
    	  System.out.println("2.AddatIndex");
    	  System.out.println("3.Remove");
    	  System.out.println("4.PrintList");
    	  System.out.println("5.Exit");
    	  
    	  int ch;
    	  System.out.println("Enter your choice");
    	  ch=sc.nextInt();
    	  
	    	  switch(ch) {
			    	  case 1:System.out.println("Enter element to push");
			    	  	
			    	  		li.add(sc.nextInt());
			    	  		break;
			    	  case 2:
			    		  System.out.println("Enter at position to add:");
			    		  int pos=sc.nextInt();
			    		  System.out.println("Enter elemnt to add:");
			    		  li.add(pos,sc.nextInt());
			    		  break;
			    	
			    	  case 3:
			    		  System.out.println("Enter position:");
			    		  
			    		  li.remove(sc.nextInt());
			    		  break;
			    		  
			    	  case 4: System.out.println("Linkedlist is:"+li);
			    	  			break;
			    	  case 5:System.exit(0);
			    	  		break;
			    	  default:
			    		  System.out.println("Your enter wrong choice");
	    	  }
    	  System.out.println("=========================");
    	 
    	  }
        }
   }
